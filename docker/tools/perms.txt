RUN addgroup --gid 1024 cmaasgroup \
   && adduser --disabled-password --gecos '' --ingroup cmaasgroup cmaas \
   && mkdir -p /cdr /cdr/inputs /cdr/outputs /cdr/temps /cdr/repos \
   && chown -R :cmaasgroup /cdr \
   && chmod -R ug=rwX /cdr \
   && chmod -R g+s /cdr

VOLUME /cdr/outputs

WORKDIR /cdr/outputs

USER cmaas

ENV PROJ_LIB=/opt/conda/share/proj
